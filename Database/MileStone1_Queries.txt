CREATE SCHEMA GamesHub;
USE GamesHub;

CREATE TABLE Buyer (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(255),
    ContactNumber VARCHAR(15),
    Email_Address VARCHAR(255) UNIQUE,
    Address VARCHAR(255),
    Password VARCHAR(72),
    Balance FLOAT
);
CREATE TABLE Seller (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Vat_Registration_Number VARCHAR(255),
    Date_Joined DATE,
    Name VARCHAR(255),
    ContactNumber VARCHAR(15),
    Email_Address VARCHAR(255) UNIQUE,
    Password VARCHAR(72),
    SellerDescription TEXT,
    Balance FLOAT,
    NationalID VARCHAR(20),
    Address VARCHAR(255)

);

CREATE TABLE Admin (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(255),
    Password VARCHAR(72),
    ContactNumber VARCHAR(15),
    EmailAddress VARCHAR(255) UNIQUE

);
